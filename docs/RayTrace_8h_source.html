163 flip = -1;

182 V dz = fr_in->dz[i];
183 //printf("%d\n", i);
184
185 while (check > epsilon)
186 {
187 t1 = refls.gp(_t, x, y, z, dx, dy, dz);
188
189 check = fabs(t1 - _t);
190
191 _t = t1;
192 }
193
194 if ((abs(round(dx)) == 0 && abs(round(dy)) == 0 && abs(round(dz)) == 0) || std::isnan(_t))
195 {
196 fr_out->x[i] = x;
197 fr_out->y[i] = y;
198 fr_out->z[i] = z;

200 fr_out->dx[i] = 0; // Set at 2: since beta should be normalized, can select on 2
201 fr_out->dy[i] = 0;
202 fr_out->dz[i] = 0;
203 }
204
205 else

207 fr_out->x[i] = x + _t*dx;

209 fr_out->z[i] = z + _t*dz;
210
211 norms = refls.np(fr_out->x[i], fr_out->y[i], fr_out->z[i], flip);

481template <class T, class U, class V>
482void RayTracer<T, U, V>::parallelRays(T ctp, U *fr_in, U *fr_out, V t0)
483{
484 int final_step;
485
486 // Transform to reflector rest frame
487 bool inv = true;
488 transfRays(ctp, fr_in, inv);
489
490 for(int n=0; n<numThreads; n++)
491 {
492 if(n == (numThreads-1)) {final_step = nTot;}
493 else {final_step = (n+1) * step;}
494
495 if (ctp.type == 0)
496 {
497 threadPool[n] = std::thread(&RayTracer::propagateRaysToP,
498 this, n * step, final_step,
499 ctp, fr_in, fr_out, t0);
500 }
501
502 else if (ctp.type == 1)
503 {
504 threadPool[n] = std::thread(&RayTracer::propagateRaysToH,
505 this, n * step, final_step,
506 ctp, fr_in, fr_out, t0);
507 }
508
509 else if (ctp.type == 2)
510 {
511 threadPool[n] = std::thread(&RayTracer::propagateRaysToE,
512 this, n * step, final_step,
513 ctp, fr_in, fr_out, t0);
514 }
515
516 else if (ctp.type == 3)
517 {
518 threadPool[n] = std::thread(&RayTracer::propagateRaysToPl,
519 this, n * step, final_step,
520 ctp, fr_in, fr_out, t0);
521 }
522 }
523 joinThreads();
524
525 // Transform back to real frame
526 transfRays(ctp, fr_in);
527 transfRays(ctp, fr_out);
528}
