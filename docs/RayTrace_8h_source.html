484 void RayTracer<T, U, V>::parallelRays(T ctp, U *fr_in, U *fr_out, V t0)
485 {

495  else {final_step = (n+1) * step;}

497  if (ctp.type == 0)

499  threadPool[n] = std::thread(&RayTracer::propagateRaysToP,

502  }
503 
504  else if (ctp.type == 1)
505  {
506  threadPool[n] = std::thread(&RayTracer::propagateRaysToH,
507  this, n * step, final_step,
508  ctp, fr_in, fr_out, t0);
509  }
510

512  {

514  this, n * step, final_step,
