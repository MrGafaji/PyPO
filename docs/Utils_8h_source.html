488{

654template <typename T> inline
655void Utils<T>::matRot(const std::array<T, 3> &rot, const std::array<T, 3> &v1, const std::array<T, 3> &cRot, std::array<T, 3> &out)
656{
657 T cosx = cos(rot[0]);
658 T cosy = cos(rot[1]);
659 T cosz = cos(rot[2]);
660
661 T sinx = sin(rot[0]);
662 T siny = sin(rot[1]);
663 T sinz = sin(rot[2]);
664
665 T mu = cosz * siny;
666 T rho = sinz * siny;
667
668 std::array<T, 3> to_rot;
669
670 for(int n=0; n<3; n++) {to_rot[n] = v1[n] - cRot[n];}
671
672 out[0] = cosz*cosy * to_rot[0] + (mu*sinx - sinz*cosx) * to_rot[1] + (mu*cosx + sinz*sinx) * to_rot[2];
673 out[1] = sinz*cosy * to_rot[0] + (rho*sinx + cosz*cosx) * to_rot[1] + (rho*cosx - cosz*sinx) * to_rot[2];
674 out[2] = -siny * to_rot[0] + cosy*sinx * to_rot[1] + cosy*cosx * to_rot[2];
675
676 for(int n=0; n<3; n++) {out[n] = out[n] + cRot[n];}
