102void RayTracer<T, U, V>::propagateRaysToP(int start, int stop,
103 T ctp, U *fr_in, U *fr_out, V t0)
104{
105 RTRefls<V> refls(ctp.coeffs[0], ctp.coeffs[1], ctp.coeffs[2]);
106
107 int flip = 1;
108
109 if (ctp.flip)
110 {
111 flip = -1;
112 }
113
114 std::array<V, 3> norms;
115
116 int jc = 0; // Counter
117 for (int i=start; i<stop; i++)
118 {
119 V _t = t0;
120 V t1 = 1e99;
121
122 V check = fabs(t1 - _t);
123
124 V x = fr_in->x[i];
125 V y = fr_in->y[i];
126 V z = fr_in->z[i];
127
128 V dx = fr_in->dx[i];
129 V dy = fr_in->dy[i];
130 V dz = fr_in->dz[i];
131 //printf("%d\n", i);
132
133 while (check > epsilon)
134 {
135 t1 = refls.gp(_t, x, y, z, dx, dy, dz);
136
137 check = fabs(t1 - _t);
138
139 _t = t1;
140 }
141 fr_out->x[i] = x + _t*dx;
142 fr_out->y[i] = y + _t*dy;
143 fr_out->z[i] = z + _t*dz;
144
145 norms = refls.np(fr_out->x[i], fr_out->y[i], fr_out->z[i], flip);
146 check = (dx*norms[0] + dy*norms[1] + dz*norms[2]);
147
148 fr_out->dx[i] = dx - 2*check*norms[0];
149 fr_out->dy[i] = dy - 2*check*norms[1];
150 fr_out->dz[i] = dz - 2*check*norms[2];
151
152 if((i * 100 / this->step) > jc && start == 0 * this->step)
153 {
154 std::cout << jc << " / 100" << '\r';
155 std::cout.flush();
156 jc++;
157 }
158
159 }

315template <class T, class U, class V>
316void RayTracer<T, U, V>::parallelRays(T ctp, U *fr_in, U *fr_out, V t0)
317{
318 int final_step;
319
320 // Transform to reflector rest frame
321 bool inv = true;
322 transfRays(ctp, fr_in, inv);
323
324 for(int n=0; n<numThreads; n++)
325 {
326 if(n == (numThreads-1)) {final_step = nTot;}
327 else {final_step = (n+1) * step;}
328
329 if (ctp.type == 0)
330 {
331 threadPool[n] = std::thread(&RayTracer::propagateRaysToP,
332 this, n * step, final_step,
333 ctp, fr_in, fr_out, t0);
334 }
335
336 else if (ctp.type == 1)
337 {
338 threadPool[n] = std::thread(&RayTracer::propagateRaysToH,
339 this, n * step, final_step,
340 ctp, fr_in, fr_out, t0);
341 }
342
343 else if (ctp.type == 2)
344 {
345 threadPool[n] = std::thread(&RayTracer::propagateRaysToE,
346 this, n * step, final_step,
347 ctp, fr_in, fr_out, t0);
348 }
349
350 else if (ctp.type == 3)
351 {
352 threadPool[n] = std::thread(&RayTracer::propagateRaysToPl,
353 this, n * step, final_step,
354 ctp, fr_in, fr_out, t0);
355 }
356 }
357 joinThreads();
358
359 // Transform back to real frame
360 transfRays(ctp, fr_in);
361 transfRays(ctp, fr_out);
362}
